Получение вью модели

```kotlin
class DeliveryPointsFragment :
    BaseRoutingFragment() {

    private var binding: FragmentDeliveryPointsBinding? = null
    private lateinit var deliveryPointsPagingAdapter: DeliveryPointsPagingAdapter

    val viewModel: RouteFragmentViewModel by lazy {
        (requireParentFragment() as RouteFragment).viewModel
    }

    override fun onCreateView(
        inflater: LayoutInflater,
        container: ViewGroup?,
        savedInstanceState: Bundle?
    ): View {
        binding = FragmentDeliveryPointsBinding.inflate(inflater, container, false)
        return binding!!.root
    }

    override fun onViewCreated(view: View, savedInstanceState: Bundle?) {
        initViews()
        bindViewModel()
    }

    private fun initViews() {
        deliveryPointsPagingAdapter = DeliveryPointsPagingAdapter(viewModel::onClickDeliveryPoint)
        binding?.recyclerView?.adapter = deliveryPointsPagingAdapter
    }

    private fun bindViewModel() {
        lifecycleScope.launch {
            viewModel.getDeliveryPointsPagingFlow.flowWithLifecycle(lifecycle, Lifecycle.State.RESUMED)
                .filterNotNull()
                .collectLatest {
                    deliveryPointsPagingAdapter.submitData(it)
                }
        }
    }

    fun setSelectedItem(deliveryPointId: Int) {
        deliveryPointsPagingAdapter.setSelectedItem(deliveryPointId)
    }

    fun clearSelectedItems() {
        deliveryPointsPagingAdapter.clearSelection()
    }

    override fun onDestroyView() {
        super.onDestroyView()
        binding = null
    }
}
```
