public class GroupTileView {

    // 0 - no incidents and non-all approved
    // 1 - all approved
    // 2 - have incidents
    private int checkApprovedAndIncidents() {
        int approvedCount = 0;
        incidentCount = 0;
        notCheckedCount = 0;
        for (Comment com : commentsOfGroup) {
            if (isTaskApproved(com)) approvedCount += 1;
            else if (isIncident(com)) incidentCount += 1;
            else notCheckedCount += 1;
        }
        if (commentsOfGroup.size() == approvedCount) return 1;
        else if (incidentCount > 0) return 2;
        else return 0;
    }
}
